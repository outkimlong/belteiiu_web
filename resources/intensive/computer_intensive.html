អស់កង្វល់ពីអនក្ខរកម្មកុំព្យូទ័រទៀតហើយ!

ថ្នាក់បំប៉នកុំព្យូទ័រ ត្រូវបានបង្កើតឡើង ដើម្បីជាជំនួយផ្នែកបច្ចេកវិទ្យាព័ត៌មានដល់និស្សិត ដើម្បីឱ្យទាន់យុគសម័យ​ឌីជីថល។ កុំព្យូទ័រ គឺជាជំនាញដ៏ចាំបាច់បំផុត សម្រាប់និស្សិតក្នុងការសិក្សាស្រាវ និងជាពិសេសសម្រាប់​ធ្វើកិច្ចការផ្សេងៗ ដែលសាស្ត្រាចារ្យដាក់ឱ្យ។ ថ្នាក់បំប៉នកុំព្យូទ័រ សិក្សារយៈពេល៦ខែ ស្មើនឹង២៥ក្រេឌីត។ និស្សិតអាចជ្រើសរើសវេនសិក្សា ដែលមិនជាន់ជាមួយនឹងវេនសិក្សាឆ្នាំទី១។ ពេលបញ្ចប់ការសិក្សាដោយជោគជ័យ សាកលវិទ្យាល័យនឹងចេញវិញ្ញាបនបត្របញ្ជាក់ការសិក្សាជូន។

❖ លក្ខខណ្ឌទូទៅ

១. បេក្ខជនទាំងពីរភេទ ជានិស្សិតខ្មែរ ឬបរទេស ព្រះសង្ឃ មន្រ្តីរាជការ បុគ្គលិកក្រុមហ៊ុន និងប្រជាពលរដ្ឋទូទៅ ដែលកំពុងសិក្សាថ្នាក់បរិញ្ញាបត្រ ឬបរិញ្ញាបត្ររង ឆ្នាំទី១ ទាំងកម្មវិធីសិក្សាជាភាសាជាតិ និងភាសាអន្តរជាតិ នៅក្នុងមហាវិទ្យាល័យណាមួយ នៃសាកលវិទ្យាល័យ ប៊ែលធី អន្តរជាតិ អាចចុះឈ្មោះចូលរៀន​ថ្នាក់បំប៉នកុំព្យូទ័រ​បាន ​ដោយឥតគិតថ្លៃ។

២. បេក្ខជន ដែលមិនមែនជានិស្សិតកំពុងសិក្សាថ្នាក់បរិញ្ញាបត្រ ឬបរិញ្ញាបត្ររង ឆ្នាំទី១ នៅក្នុងមហាវិទ្យាល័យណាមួយ នៃសាកលវិទ្យាល័យ ប៊ែលធី អន្តរជាតិ អាចចុះឈ្មោះចូលរៀនបាន ដោយតម្រូវឱ្យបង់ប្រាក់ថ្លៃសិក្សា ទៅតាមតារាងតម្លៃដែលបានកំណត់។

៣. ចំពោះនិស្សិត ដែលកំពុងសិក្សាឆ្នាំទី២ ទី៣ ឬទី៤ ដែលត្រូវការបំប៉នកុំព្យូទ័របន្ថែម ក៏អាចចុះឈ្មោះចូលរៀនបាន ដោយតម្រូវឱ្យបង់ប្រាក់ថ្លៃសិក្សា ទៅតាមតារាងតម្លៃដែលបានកំណត់ និងដោយមានការអនុញ្ញាតពីឯកឧត្តមសាកលវិទ្យាធិការ ឬឯកឧត្តមបណ្ឌិត ប្រធានក្រុមប្រឹក្សាភិបាល។
